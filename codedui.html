Hope you have installed chili first before starting off.

use the following namespaces

                 
             using Microsoft.VisualStudio.TestTools.UnitTesting;
             using System.Data;
             using System.Collections.Generic;
             using System.Linq;
             using Chili;

Create a class inside your project and name it Report.cs and associate it with attribute [TestClass] as well. Declare followings variables.

             
              private static DataTable DataTable = null;
              private static int Passed = 0;

Create a method with [AssemblyInitialize] 

             [AssemblyInitialize]
             public static void CreateTableFormat(TestContext t)
             {
                 DataTable = new DataTable();
                 DataTable.Columns.Add("Feature");
                 DataTable.Columns.Add("Test Case Name");
                 DataTable.Columns.Add("Outcome");
             }

Create a method as below with no attribute

                 
              public static void CreateReport(string feature,string testcaseName,string outcome)
              {
                  DataRow newrow = DataTable.NewRow();
                  newrow[0] = feature;
                  newrow[1] = testcaseName;
                  newrow[2] = outcome;
                  if (outcome.Equals("Passed"))
                     Passed++;
                  DataTable.Rows.Add(newrow);
              }

Create a method with [AssemblyCleanup]
[TestCleanup()] public void MyTestCleanup() { Report.CreateReport(this.GetType().Name,"TestContext.TestName","Failed"); }
